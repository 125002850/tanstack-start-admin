# No environment variables required for basic usage

# Gateway prefix used by Vite proxy and OpenAPI codegen transport basePath.
# APP_GATEWAY=/admin-api

# Backend API target used by the Vite dev server proxy.
# PROXY_URL=http://localhost:8080

# OpenAPI document URL used by codegen. Override when a local backend runs on a
# non-default port.
# OPENAPI_FETCH_TARGET=http://localhost:8080/v3/api-docs

# Frontend public base path used by Vite asset URLs and TanStack Router.
# APP_BASE_PATH=/admin

# Enable workspace tabs feature (defaults to enabled when unset).
# Set to 0 to fully fall back to legacy page-cache / PageContainer rendering.
# VITE_ENABLE_WORKSPACE_TABS=1

# Enable DataTable virtual scrolling (defaults to enabled when unset).
# Set to 0 to disable virtual scrolling for all data tables.
# VITE_ENABLE_DATA_TABLE_VIRTUALIZATION=1
# (Legacy) VITE_ENABLE_PRODUCT_TABLE_VIRTUALIZATION — fallback if VITE_ENABLE_DATA_TABLE_VIRTUALIZATION is unset
